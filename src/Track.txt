/**
 * class Track
 *
 * @author Jim
 * @version 1.0
 */
public class Track 
{
    // instance variables
    private String title;
    private double time;
    
    /**
     * Default constructor
     */
    public Track() {
        title = "Untitled";
        time = 0.0;
    }
    
    /**
     * Constructor using title and time
     */
    public Track(String title, double time) {
        this.title = title;
        this.time = time;

    
    /**
     * Accessors and mutators
     *
     */
    public void setTitle(String title) {
        this.title = title;
    }
    public String getTitle() {
        return(title);
    }
    public void setTime(double time) {
        this.time = time;
    }
    public double getTime() {
        return(time);
    }
}
